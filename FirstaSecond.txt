#include <iostream>
#include <cstring>
#include <fstream>

using std::cout;
using std::cin;
using std::ios;
using std::fstream;

class reverse
{
    fstream ifile;
    fstream ofile;
    char buff[40];

public:
    reverse(){
        cout << "Enter input File name :";
        cin >> buff;
        ifile.open(buff,ios::in);

        cout << "Enter output File name :";
        cin >> buff;
        ofile.open(buff,ios::out | ios::trunc);
    }

    void run(){

        while(!ifile.eof()){
            ifile.getline(buff,41);
            //_strrev(buff);
            if(strlen(buff)){
                strcat(buff, "\n");
                ofile.write(buff,strlen(buff));
            }
        }
    }

};

int main(void){
    reverse reverse;
    reverse.run();
}
